START
TEMP = 0
DIGIT = 0
RESULT = 0
INPUT NUM

WHILE NUM > 0 DO
DIGIT = NUM % 10

IF TEMP == 9 && DIGIT == 0 THEN
NUM = NUM / 100
ELSE
RESULT = RESULT * 10 + DIGIT
TEMP = DIGIT
NUM = NUM / 10
END IF
END WHILE

PRINT "Number: ", RESULT
END
