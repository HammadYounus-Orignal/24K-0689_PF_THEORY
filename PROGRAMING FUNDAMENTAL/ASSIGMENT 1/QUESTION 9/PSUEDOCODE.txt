START 

INT AGEDAYS = 0 , AGEMONTHS = 0 , AGEYEARS = 0

PRINT " INPUT BIRTHDAY "
INPUT BIRTHDAY
PRINT " INPUT BIRTHMONTH "
INPUT BIRTHMONTH
PRINT " INPUT BIRTHYEAR "
INPUT BIRTHYEAR
PRINT " INPUT CURRENTDAY "
INPUT CURRENTDAY
PRINT " INPUT CURRENTMONTH "
INPUT CURRENTMONTH
PRINT " INPUT CURRENTYEAR "
INPUT CURRENTYEAR

IF BIRTHMONTH < 1 || BIRTHMONTH > 12 THEN
PRINT " INVAILD MONTH "
ELSE IF BIRTHDAY < 1 || BIRTHDAY > 31 THEN
PRINT " INVALID BIRTHDAY "
ELSE IF BIRTHYEAR < 1 THEN
PRINT " INVAILD BIRTHYEAR "
ELSE IF CURRENTMONTH < 1 || CURRENTMONTH > 12 THEN
PRINT " INVALID CURRENTMONTH "
ELSE IF CURRENTDAY < 1 || CURRENTDAY > 31 THEN
PRINT "INVAIL CURRENTDAY"
ELSE IF CURRENTYEAR < BIRTHYEAR THEN
PRINT "CURRENT YEAR IS NOT VALID"
ELSE
AGEDAYS = (CURRENTYEAR - BIRTHYEAR) * 365 + (CURRENTMONTH - BIRTHMONTH) * 30 + (CURRENTDAY - BIRTHDAY)
AGEMONTHS = AGEDAYS / 30
AGEYEARS = AGEDAYS / 365
END IF 

PRINT " AGE IN DAYS : " AGEDAYS
PRINT " AGE IN MONTHS : " AGEMONTHS
PRINT " AGE IN YEARS : " AGEYEARS

END
